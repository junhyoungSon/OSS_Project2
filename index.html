Java: 간단한 토큰 카운터

// File: TokenCounter.java
// WHY: 공백/구두점 기준으로 토큰을 분리해 개수 계산 (간단 검증용)
import java.util.*;
import java.util.regex.*;

public class TokenCounter {
    private static final Pattern TOKEN = Pattern.compile("[\가-\힯\w]+", Pattern.UNICODE_CHARACTER_CLASS);

    public static int countTokens(String text) {
        if (text == null || text.isEmpty()) return 0;
        Matcher m = TOKEN.matcher(text);
        int c = 0; while (m.find()) c++;
        return c;
    }

    public static void main(String[] args) {
        String s = String.join(" ", Arrays.asList(args));
        System.out.println(countTokens(s));
    }
}

C: 누적합(prefix sum) - 대용량 처리용

// File: prefix_sum.c
// WHY: 메모리 접근 패턴 단순화로 캐시 효율 개선
#include <stdio.h>
#include <stdlib.h>

int prefix_sum(const int *in, int *out, size_t n) {
    if (!in || !out) return -1; // WHY: 안전성
    long long acc = 0;          // WHY: 오버플로 완화(입력 범위 넉넉히)
    for (size_t i = 0; i < n; ++i) {
        acc += in[i];
        out[i] = (int)acc;
    }
    return 0;
}

int main(void){
    int n; if (scanf("%d", &n) != 1) return 1;
    int *a = malloc(sizeof(int)*n);
    int *b = malloc(sizeof(int)*n);
    for (int i=0;i<n;i++) scanf("%d", &a[i]);
    prefix_sum(a,b,n);
    for (int i=0;i<n;i++) printf("%d%c", b[i], i+1==n?'\n':' ');
    free(a); free(b);
    return 0;
}
